async function buscarMuralEHorarios() { const divAvisos = document.getElementById('lista-avisos'); const divHorarios = document.getElementById('lista-horarios'); // --- BLOCO 1: BUSCAR AVISOS/MENSAGENS --- if(divAvisos) { try { const qAvisos = query(collection(db, "mensagens"), where("equipe_id", "==", alunoLogado.equipe_id)); const snapAvisos = await getDocs(qAvisos); let avisos = []; const hojeIso = new Date().toISOString(); snapAvisos.forEach(doc => { const av = doc.data(); if (av.status === 'inativo') return; if (av.data_expiracao && av.data_expiracao < hojeIso) return; avisos.push(av); }); avisos.sort((a,b) => { let dateA = a.data_envio ? new Date(a.data_envio).getTime() : 0; let dateB = b.data_envio ? new Date(b.data_envio).getTime() : 0; return dateB - dateA; }); if(avisos.length > 0) { divAvisos.innerHTML = avisos.map(av => { let corBorda = 'border-slate-600/50'; let corIcone = 'text-cyan-400'; let icone = '📌'; if (av.urgencia === 'urgente') { corBorda = 'border-rose-500/50'; corIcone = 'text-rose-400'; icone = '🚨'; } else if (av.urgencia === 'importante') { corBorda = 'border-amber-500/50'; corIcone = 'text-amber-400'; icone = '⚠️'; } return `
${icone}

${av.titulo || 'Aviso'}

${av.conteudo || ''}

`; }).join(''); } else { divAvisos.innerHTML = "
Nenhum recado no momento.
"; } } catch (error) { console.error("Erro Avisos:", error); divAvisos.innerHTML = "
Erro ao carregar avisos.
"; } } // --- BLOCO 2: BUSCAR HORÁRIOS (DIRETO DAS TURMAS) --- if(divHorarios) { try { const qTurmas = query(collection(db, "turmas"), where("equipe_id", "==", alunoLogado.equipe_id)); const snapTurmas = await getDocs(qTurmas); let turmas = []; snapTurmas.forEach(doc => { const t = doc.data(); turmas.push({ nome: t.nome_turma || t.nome || t.titulo || 'Treino', dias_array: Array.isArray(t.dias_semana) ? t.dias_semana : [], // NOVO FORMATO ARRAY hora: t.hora_inicio || t.horario || '--:--', isMinhaTurma: (t.alunos_ids && t.alunos_ids.includes(alunoLogado.id)) }); }); // FUNÇÃO PARA CALCULAR PRÓXIMA DATA function getProximoTreino(diasArray, horaInicio) { if(!diasArray || diasArray.length === 0 || !horaInicio || horaInicio === '--:--') return null; const agora = new Date(); const diaHoje = agora.getDay(); const partesHora = horaInicio.split(':'); if(partesHora.length !== 2) return null; const hora = parseInt(partesHora[0]); const min = parseInt(partesHora[1]); let menorDifDias = 99; let dataAlvo = null; diasArray.forEach(d => { let dif = d - diaHoje; if(dif < 0) dif += 7; // Se for hoje, verifica se já passou do horário if(dif === 0) { const dataHojeTeste = new Date(agora); dataHojeTeste.setHours(hora, min, 0, 0); if(agora > dataHojeTeste) { dif = 7; // Próxima semana } } if(dif < menorDifDias) { menorDifDias = dif; const proxDate = new Date(agora); proxDate.setDate(agora.getDate() + dif); proxDate.setHours(hora, min, 0, 0); dataAlvo = proxDate; } }); return dataAlvo; } const nomesDias = ['Domingo', 'Segunda', 'Terça', 'Quarta', 'Quinta', 'Sexta', 'Sábado']; // Aplica o cálculo em todas as turmas turmas.forEach(t => { t.proximaSessao = getProximoTreino(t.dias_array, t.hora); }); // Verifica se o aluno tem turmas dele. Se tiver, filtra SÓ as dele. Senão, mostra todas. let temTurmaAtleta = turmas.some(t => t.isMinhaTurma); let turmasExibidas = temTurmaAtleta ? turmas.filter(t => t.isMinhaTurma) : turmas; // Ordena as turmas pela próxima sessão cronologicamente! turmasExibidas.sort((a,b) => (a.proximaSessao || 0) - (b.proximaSessao || 0)); if(turmasExibidas.length > 0) { divHorarios.innerHTML = turmasExibidas.map(t => { let labelDia = '--'; let iconeDia = '📅'; let corDia = 'text-slate-400'; if(t.proximaSessao) { const hoje = new Date(); const prox = t.proximaSessao; const isHoje = hoje.getDate() === prox.getDate() && hoje.getMonth() === prox.getMonth() && hoje.getFullYear() === prox.getFullYear(); const amanha = new Date(); amanha.setDate(hoje.getDate() + 1); const isAmanha = amanha.getDate() === prox.getDate() && amanha.getMonth() === prox.getMonth(); if(isHoje) { labelDia = 'HOJE'; iconeDia = '🟢'; corDia = 'text-emerald-400 font-black tracking-widest'; } else if(isAmanha) { labelDia = 'AMANHÃ'; iconeDia = '🟡'; corDia = 'text-amber-400 font-black tracking-widest'; } else { labelDia = nomesDias[prox.getDay()]; corDia = 'text-indigo-300 font-bold'; } } return `

${t.nome}

${iconeDia} ${labelDia}

${t.hora}
`; }).join(''); } else { divHorarios.innerHTML = "
Nenhuma turma para exibir.
"; } } catch (error) { console.error("Erro Horários/Turmas:", error); divHorarios.innerHTML = "
Erro ao carregar a grade de treinos.
"; } } }
